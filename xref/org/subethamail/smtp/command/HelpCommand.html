1   package org.subethamail.smtp.command;
2   
3   import java.io.IOException;
4   
5   import org.subethamail.smtp.server.BaseCommand;
6   import org.subethamail.smtp.server.CommandException;
7   import org.subethamail.smtp.server.SMTPServer;
8   import org.subethamail.smtp.server.Session;
9   
10  /**
11   * Provides a help topic system for people to interact with.
12   *
13   * @author Ian McFarland &lt;ian@neo.com&gt;
14   * @author Jon Stevens
15   * @author Scott Hernandez
16   */
17  public class HelpCommand extends BaseCommand {
18  	public HelpCommand() {
19  		super("HELP",
20  				"The HELP command gives help info about the topic specified.\n"
21  						+ "For a list of topics, type HELP by itself.",
22  				"[ <topic> ]");
23  	}
24  
25  	@Override
26  	public void execute(final String commandString, final Session context) throws IOException {
27  		final String args = this.getArgPredicate(commandString);
28  		if ("".equals(args)) {
29  			context.sendResponse(this.getCommandMessage(context.getServer()));
30  			return;
31  		}
32  		try {
33  			context.sendResponse(context.getServer().getCommandHandler().getHelp(args).toOutputString());
34  		} catch (final CommandException e) {
35  			context.sendResponse("504 HELP topic \"" + args + "\" unknown.");
36  		}
37  	}
38  
39  	private String getCommandMessage(final SMTPServer server) {
40  		return "214-"
41  				+ server.getSoftwareName()
42  				+ " on "
43  				+ server.getHostName()
44  				+ "\r\n"
45  				+ "214-Topics:\r\n"
46  				+ this.getFormattedTopicList(server)
47  				+ "214-For more info use \"HELP <topic>\".\r\n"
48  				+ "214 End of HELP info";
49  	}
50  
51  	protected String getFormattedTopicList(final SMTPServer server) {
52  		final StringBuilder sb = new StringBuilder();
53  		for (final String key : server.getCommandHandler().getVerbs()) {
54  			sb.append("214-     " + key + "\r\n");
55  		}
56  		return sb.toString();
57  	}
58  }
