1   package org.subethamail.smtp.command;

3   import java.io.IOException;
4   import java.util.Locale;
5   
6   import org.subethamail.smtp.DropConnectionException;
7   import org.subethamail.smtp.RejectException;
8   import org.subethamail.smtp.server.BaseCommand;
9   import org.subethamail.smtp.server.Session;
10  import org.subethamail.smtp.util.EmailUtils;
11  
12  /**
13   * @author Ian McFarland &lt;ian@neo.com&gt;
14   * @author Jon Stevens
15   * @author Jeff Schnitzer
16   */
17  public class ReceiptCommand extends BaseCommand {
18  	public ReceiptCommand() {
19  		super("RCPT", "Specifies the recipient. Can be used any number of times.", "TO: <recipient> [ <parameters> ]");
20  	}
21  
22  	@Override
23  	public void execute(final String commandString, final Session sess) throws IOException, DropConnectionException {
24  		if (!sess.isMailTransactionInProgress()) {
25  			sess.sendResponse("503 5.5.1 Error: need MAIL command");
26  			return;
27  		}
28  		if (sess.getServer().getMaxRecipients() >= 0
29  				&& sess.getRecipientCount() >= sess.getServer().getMaxRecipients()) {
30  			sess.sendResponse("452 Error: too many recipients");
31  			return;
32  		}
33  
34  		final String args = this.getArgPredicate(commandString);
35  		if (!args.toUpperCase(Locale.ENGLISH).startsWith("TO:")) {
36  			sess.sendResponse("501 Syntax: RCPT TO: <address>  Error in parameters: \"" + args + "\"");
37  		}
38  		final String recipientAddress = EmailUtils.extractEmailAddress(args, 3);
39  		try {
40  			sess.getMessageHandler().recipient(recipientAddress);
41  			sess.addRecipient(recipientAddress);
42  			sess.sendResponse("250 Ok");
43  		} catch (final DropConnectionException ex) {
44  			throw ex; // Propagate this
45  		} catch (final RejectException ex) {
46  			sess.sendResponse(ex.getErrorResponse());
47  		}
48  	}
49  }
