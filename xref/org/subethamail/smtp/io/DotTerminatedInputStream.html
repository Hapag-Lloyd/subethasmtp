1   package org.subethamail.smtp.io;
2   
3   import java.io.EOFException;
4   import java.io.FilterInputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   
8   /**
9    * An InputStream class that terminates the stream when it encounters a US-ASCII
10   * encoded dot CR LF byte sequence immediately following a CR LF line end.
11   */
12  public class DotTerminatedInputStream extends FilterInputStream {
13  	/**
14  	 * The last bytes returned by the {@link #read()} function. The first byte in
15  	 * the array contains the byte returned by the penultimate read() call. The
16  	 * second byte in the array contains the byte returned by the last read() call.
17  	 * EOF (-1) is not shifted into the array. It's initial value is CR LF, so the
18  	 * first character of the stream is considered to be the first character of a
19  	 * line. This makes it possible to receive empty data.
20  	 */
21  	private final byte[] lastBytes = new byte[] { '\r', '\n' };
22  
23  	/**
24  	 * The buffer which contains the bytes read from the underlying stream in
25  	 * advance. These bytes are not yet returned by the {@link #read()} function.
26  	 * Null means uninitialized.
27  	 */
28  	private int[] nextBytes = null;
29  
30  	/**
31  	 * Indicates that the last byte - not including the terminating sequence - of
32  	 * the wrapped stream was already returned by {@link #read()}
33  	 */
34  	private boolean endReached = false;
35  
36  	/**
37  	 * A constructor for this object that takes a stream to be wrapped and a
38  	 * terminating character sequence.
39  	 *
40  	 * @param in the <code>InputStream</code> to be wrapped
41  	 * @throws IllegalArgumentException if the terminator array is null or empty
42  	 */
43  	public DotTerminatedInputStream(final InputStream in) {
44  		super(in);
45  	}
46  
47  	@Override
48  	public int read() throws IOException {
49  		if (nextBytes == null) {
50  			initNextBytes();
51  		}
52  		if (endReached) {
53  			return -1;
54  		}
55  		if (lastBytesAreCrLf() && nextBytesAreDotCrLf()) {
56  			endReached = true;
57  			return -1;
58  		}
59  		final int result = nextBytes[0];
60  		if (result == -1) {
61  			// End of stream reached without seeing the terminator
62  			throw new EOFException("Pre-mature end of <CRLF>.<CRLF> terminated data");
63  		}
64  		readWrappedStream();
65  		return result;
66  	}
67  
68  	private void initNextBytes() throws IOException {
69  		nextBytes = new int[3];
70  		nextBytes[0] = super.read();
71  		nextBytes[1] = super.read();
72  		nextBytes[2] = super.read();
73  	}
74  
75  	private boolean lastBytesAreCrLf() {
76  		return lastBytes[0] == '\r' && lastBytes[1] == '\n';
77  	}
78  
79  	private boolean nextBytesAreDotCrLf() {
80  		return nextBytes[0] == '.' && nextBytes[1] == '\r' && nextBytes[2] == '\n';
81  	}
82  
83  	/**
84  	 * Shifts bytes in the buffers, reads a byte from the wrapped stream, and places
85  	 * it at the end of the nextBytes buffer.
86  	 */
87  	private void readWrappedStream() throws IOException {
88  		lastBytes[0] = lastBytes[1];
89  		// casting is safe, this function is not called if an - unexpected - EOF
90  		// was read
91  		lastBytes[1] = (byte) nextBytes[0];
92  		nextBytes[0] = nextBytes[1];
93  		nextBytes[1] = nextBytes[2];
94  		nextBytes[2] = super.read();
95  	}
96  }
