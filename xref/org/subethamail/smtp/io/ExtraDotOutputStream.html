15  package org.subethamail.smtp.io;
16  
17  import java.io.IOException;
18  import java.io.OutputStream;
19  
20  /**
21   * Adds extra dot if dot occurs in message body at beginning of line (according
22   * to RFC1939) Compare also org.apache.james.smtpserver.SMTPInputStream
23   */
24  public class ExtraDotOutputStream extends CRLFOutputStream {
25  	/**
26  	 * Constructor that wraps an OutputStream.
27  	 *
28  	 * @param out the OutputStream to be wrapped
29  	 */
30  	public ExtraDotOutputStream(final OutputStream out) {
31  		super(out);
32  	}
33  
34  	/**
35  	 * Overrides super writeChunk in order to add a "." if the previous chunk ended
36  	 * with a new line and a new chunk starts with "."
37  	 *
38  	 * @see CRLFOutputStream#writeChunk(byte[], int, int)
39  	 */
40  	@Override
41  	protected void writeChunk(final byte buffer[], final int offset, final int length) throws IOException {
42  		if (length > 0 && buffer[offset] == '.' && this.startOfLine) {
43  			// add extra dot (the first of the pair)
44  			this.out.write('.');
45  		}
46  		super.writeChunk(buffer, offset, length);
47  	}
48  
49  	/**
50  	 * Writes a byte to the stream, adding dots where appropriate. Also fixes any
51  	 * naked CR or LF to the RFC 2821 mandated CRLF pairing.
52  	 *
53  	 * @param b the byte to write
54  	 *
55  	 * @throws IOException if an error occurs writing the byte
56  	 */
57  	@Override
58  	public void write(final int b) throws IOException {
59  		if (b == '.' && this.statusLast != LAST_WAS_OTHER) {
60  			// add extra dot (the first of the pair)
61  			this.out.write('.');
62  		}
63  		super.write(b);
64  	}
65  }
