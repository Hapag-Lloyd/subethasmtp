4   package org.subethamail.smtp.io;
5   
6   import java.io.BufferedInputStream;
7   import java.io.BufferedOutputStream;
8   import java.io.ByteArrayOutputStream;
9   import java.io.File;
10  import java.io.FileInputStream;
11  import java.io.FileOutputStream;
12  import java.io.IOException;
13  import java.io.InputStream;
14  
15  /**
16   * This works like a ByteArrayOutputStream until a certain size is reached, then
17   * creates a temp file and acts like a buffered FileOutputStream. The data can
18   * be retreived afterwards by calling getInputStream().
19   *
20   * When this object is closed, the temporary file is deleted. You can no longer
21   * call getInputStream().
22   *
23   * @author Jeff Schnitzer
24   */
25  public class DeferredFileOutputStream extends ThresholdingOutputStream {
26  	/**
27  	 * Initial size of the byte array buffer. Better to make this large to start
28  	 * with so that we can avoid reallocs; mail messages are rarely tiny.
29  	 */
30  	static final int INITIAL_BUF_SIZE = 8192;
31  
32  	public static final String TMPFILE_PREFIX = "subetha";
33  
34  	public static final String TMPFILE_SUFFIX = ".msg";
35  
36  	/** If we switch to file output, this is the file. */
37  	File outFile;
38  
39  	/** If we switch to file output, this is the stream. */
40  	FileOutputStream outFileStream;
41  
42  	/** When the output stream is closed, this becomes true */
43  	boolean closed;
44  
45  	boolean thresholdReached = false;
46  
47  	/**
48  	 * @param transitionSize is the number of bytes at which to convert from a byte
49  	 *                       array to a real file.
50  	 */
51  	public DeferredFileOutputStream(final int transitionSize) {
52  		super(new BetterByteArrayOutputStream(INITIAL_BUF_SIZE), transitionSize);
53  	}
54  
55  	/*
56  	 * (non-Javadoc)
57  	 *
58  	 * @see org.subethamail.common.io.ThresholdingOutputStream#thresholdReached(int,
59  	 * int)
60  	 */
61  	@Override
62  	protected void thresholdReached(final int current, final int predicted) throws IOException {
63  		// Open a temp file, write the byte array version, and swap the
64  		// output stream to the file version.
65  
66  		this.outFile = File.createTempFile(TMPFILE_PREFIX, TMPFILE_SUFFIX);
67  		this.outFileStream = new FileOutputStream(this.outFile);
68  
69  		((ByteArrayOutputStream) this.output).writeTo(this.outFileStream);
70  		this.output = new BufferedOutputStream(this.outFileStream);
71  	}
72  
73  	/**
74  	 * Closes the output stream and creates an InputStream on the same data.
75  	 *
76  	 * @return either a BetterByteArrayOutputStream or buffered FileInputStream,
77  	 *         depending on what state we are in.
78  	 */
79  	public InputStream getInputStream() throws IOException {
80  		if (this.output instanceof BetterByteArrayOutputStream) {
81  			return ((BetterByteArrayOutputStream) this.output).getInputStream();
82  		}
83  		if (!this.closed) {
84  			this.output.flush();
85  			this.output.close();
86  			this.closed = true;
87  		}
88  
89  		return new BufferedInputStream(new FileInputStream(this.outFile));
90  	}
91  
92  	/*
93  	 * (non-Javadoc)
94  	 *
95  	 * @see org.subethamail.common.io.ThresholdingOutputStream#close()
96  	 */
97  	@Override
98  	public void close() throws IOException {
99  		if (!this.closed) {
100 			this.output.flush();
101 			this.output.close();
102 			this.closed = true;
103 		}
104 
105 		if (this.outFile != null) {
106 			this.outFile.delete();
107 		}
108 	}
109 }
