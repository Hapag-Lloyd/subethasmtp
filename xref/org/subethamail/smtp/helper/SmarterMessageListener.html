6   package org.subethamail.smtp.helper;
7   
8   import java.io.IOException;
9   import java.io.InputStream;
10  
11  import org.subethamail.smtp.TooMuchDataException;
12  
13  /**
14   * This is an interface for processing the end-result messages that is
15   * higher-level than the MessageHandler and related factory but lower-level than
16   * SimpleMessageListener.
17   *
18   * @author Jeff Schnitzer
19   */
20  public interface SmarterMessageListener {
21  	/**
22  	 * Called once for every RCPT TO during a SMTP exchange. Each accepted recipient
23  	 * returns a Receiver which will have its deliver() mehtod called later.
24  	 *
25  	 * @param from      is a rfc822-compliant email address.
26  	 * @param recipient is a rfc822-compliant email address.
27  	 *
28  	 * @return A Receiver if the listener wants delivery of the message, null if the
29  	 *         message is not to be accepted.
30  	 */
31  	Receiver accept(String from, String recipient);
32  
33  	/**
34  	 * Interface which accepts delivery of a message.
35  	 */
36  	public interface Receiver {
37  		/**
38  		 * When message data arrives, this method will be called for every recipient
39  		 * this listener accepted.
40  		 *
41  		 * @param data will be the smtp data stream, stripped of any extra '.' chars.
42  		 *             The data stream is only valid for the duration of this call.
43  		 *
44  		 * @throws TooMuchDataException if the listener can't handle that much data. An
45  		 *                              error will be reported to the client.
46  		 * @throws IOException          if there is an IO error reading the input data.
47  		 */
48  		void deliver(InputStream data) throws TooMuchDataException, IOException;
49  
50  		/**
51  		 * Called at the end of the SMTP exchange, even if no data was delivered.
52  		 */
53  		void done();
54  	}
55  }
