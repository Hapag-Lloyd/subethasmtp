4   package org.subethamail.smtp.helper;
5   
6   import java.io.IOException;
7   import java.io.InputStream;
8   
9   import org.subethamail.smtp.TooMuchDataException;
10  
11  /**
12   * This is an interface for processing the end-result messages that is
13   * higher-level than the MessageHandler and related factory.
14   *
15   * While the SMTP message is being received, all listeners are asked if they
16   * want to accept each recipient. After the message has arrived, the message is
17   * handed off to all accepting listeners.
18   *
19   * @author Jeff Schnitzer
20   */
21  public interface SimpleMessageListener {
22  	/**
23  	 * Called once for every RCPT TO during a SMTP exchange. Each accepted recipient
24  	 * will result in a separate deliver() call later.
25  	 *
26  	 * @param from      is a rfc822-compliant email address.
27  	 * @param recipient is a rfc822-compliant email address.
28  	 *
29  	 * @return true if the listener wants delivery of the message, false if the
30  	 *         message is not for this listener.
31  	 */
32  	boolean accept(String from, String recipient);
33  
34  	/**
35  	 * When message data arrives, this method will be called for every recipient
36  	 * this listener accepted.
37  	 *
38  	 * @param from      is the envelope sender in rfc822 form
39  	 * @param recipient will be an accepted recipient in rfc822 form
40  	 * @param data      will be the smtp data stream, stripped of any extra '.'
41  	 *                  chars. The data stream is only valid for the duration of
42  	 *                  this call.
43  	 *
44  	 * @throws TooMuchDataException if the listener can't handle that much data. An
45  	 *                              error will be reported to the client.
46  	 * @throws IOException          if there is an IO error reading the input data.
47  	 */
48  	void deliver(String from, String recipient, InputStream data) throws TooMuchDataException, IOException;
49  }
