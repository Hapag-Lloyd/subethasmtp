5   import org.subethamail.smtp.DropConnectionException;

7   /**
8    * Verifies the presence of a TLS connection if TLS is required. The wrapped
9    * command is executed when the test succeeds.
10   *
11   * @author Erik van Oosten
12   */
13  public class RequireTLSCommandWrapper implements Command {
14  	private final Command wrapped;
15  
16  	/**
17  	 * @param wrapped the wrapped command (not null)
18  	 */
19  	public RequireTLSCommandWrapper(final Command wrapped) {
20  		this.wrapped = wrapped;
21  	}
22  
23  	@Override
24  	public void execute(final String commandString, final Session sess) throws IOException, DropConnectionException {
25  		if (!sess.getServer().getRequireTLS() || sess.isTLSStarted()) {
26  			wrapped.execute(commandString, sess);
27  		} else {
28  			sess.sendResponse("530 Must issue a STARTTLS command first");
29  		}
30  	}
31  
32  	@Override
33  	public HelpMessage getHelp() throws CommandException {
34  		return wrapped.getHelp();
35  	}
36  
37  	@Override
38  	public String getName() {
39  		return wrapped.getName();
40  	}
41  }
