1   package org.subethamail.smtp.server;
2   
3   import java.io.IOException;
4   import java.net.SocketTimeoutException;
5   import java.util.Collection;
6   import java.util.HashMap;
7   import java.util.Locale;
8   import java.util.Map;
9   import java.util.Set;
10  import java.util.StringTokenizer;
11  
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  import org.subethamail.smtp.DropConnectionException;
15  
16  /**
17   * This class manages execution of a SMTP command.
18   *
19   * @author Jon Stevens
20   * @author Scott Hernandez
21   */
22  public class CommandHandler {
23  	private final static Logger log = LoggerFactory.getLogger(CommandHandler.class);
24  
25  	/**
26  	 * The map of known SMTP commands. Keys are upper case names of the commands.
27  	 */
28  	private final Map<String, Command> commandMap = new HashMap<>();
29  
30  	public CommandHandler() {
31  		// This solution should be more robust than the earlier "manual" configuration.
32  		for (final CommandRegistry registry : CommandRegistry.values()) {
33  			this.addCommand(registry.getCommand());
34  		}
35  	}
36  
37  	/**
38  	 * Create a command handler with a specific set of commands.
39  	 *
40  	 * @param availableCommands the available commands (not null) TLS note: wrap
41  	 *                          commands with {@link RequireTLSCommandWrapper} when
42  	 *                          appropriate.
43  	 */
44  	public CommandHandler(final Collection<Command> availableCommands) {
45  		for (final Command command : availableCommands) {
46  			this.addCommand(command);
47  		}
48  	}
49  
50  	/**
51  	 * Adds or replaces the specified command.
52  	 */
53  	public void addCommand(final Command command) {
54  		if (log.isDebugEnabled()) {
55  			log.debug("Added command: " + command.getName());
56  		}
57  
58  		this.commandMap.put(command.getName(), command);
59  	}
60  
61  	/**
62  	 * Returns the command object corresponding to the specified command name.
63  	 *
64  	 * @param commandName case insensitive name of the command.
65  	 * @return the command object, or null, if the command is unknown.
66  	 */
67  	public Command getCommand(final String commandName) {
68  		final String upperCaseCommandName = commandName.toUpperCase(Locale.ENGLISH);
69  		return this.commandMap.get(upperCaseCommandName);
70  	}
71  
72  	public boolean containsCommand(final String command) {
73  		return this.commandMap.containsKey(command);
74  	}
75  
76  	public Set<String> getVerbs() {
77  		return this.commandMap.keySet();
78  	}
79  
80  	public void handleCommand(final Session context, final String commandString)
81  			throws SocketTimeoutException, IOException, DropConnectionException {
82  		try {
83  			final Command command = this.getCommandFromString(commandString);
84  			command.execute(commandString, context);
85  		} catch (final CommandException e) {
86  			context.sendResponse("500 " + e.getMessage());
87  		}
88  	}
89  
90  	/**
91  	 * @return the HelpMessage object for the given command name (verb)
92  	 * @throws CommandException on command error
93  	 */
94  	public HelpMessage getHelp(final String command) throws CommandException {
95  		return this.getCommandFromString(command).getHelp();
96  	}
97  
98  	private Command getCommandFromString(final String commandString)
99  			throws UnknownCommandException, InvalidCommandNameException {
100 		Command command = null;
101 		final String key = this.toKey(commandString);
102 		if (key != null) {
103 			command = this.commandMap.get(key);
104 		}
105 		if (command == null) {
106 			// some commands have a verb longer than 4 letters
107 			final String verb = this.toVerb(commandString);
108 			if (verb != null) {
109 				command = this.commandMap.get(verb);
110 			}
111 		}
112 		if (command == null) {
113 			throw new UnknownCommandException("Error: command not implemented");
114 		}
115 		return command;
116 	}
117 
118 	private String toKey(final String string) throws InvalidCommandNameException {
119 		if (string == null || string.length() < 4) {
120 			throw new InvalidCommandNameException("Error: bad syntax");
121 		}
122 
123 		return string.substring(0, 4).toUpperCase(Locale.ENGLISH);
124 	}
125 
126 	private String toVerb(final String string) throws InvalidCommandNameException {
127 		final StringTokenizer stringTokenizer = new StringTokenizer(string);
128 		if (!stringTokenizer.hasMoreTokens()) {
129 			throw new InvalidCommandNameException("Error: bad syntax");
130 		}
131 
132 		return stringTokenizer.nextToken().toUpperCase(Locale.ENGLISH);
133 	}
134 }
