1   package org.subethamail.smtp.server;

3   import java.util.Locale;
4   
5   import javax.annotation.concurrent.GuardedBy;
6   import javax.annotation.concurrent.ThreadSafe;
7   
8   /**
9    * TimeBasedSessionIdFactory is a very simple {@link SessionIdFactory}, which
10   * assigns numeric identifiers based on the current milliseconds time, amending
11   * it as necessary to make it unique.
12   */
13  @ThreadSafe
14  public class TimeBasedSessionIdFactory implements SessionIdFactory {
15  	@GuardedBy("this")
16  	private long lastAllocatedId = 0;
17  
18  	@Override
19  	public String create() {
20  		long id = System.currentTimeMillis();
21  		synchronized (this) {
22  			if (id <= lastAllocatedId) {
23  				id = lastAllocatedId + 1;
24  			}
25  			lastAllocatedId = id;
26  		}
27  		return Long.toString(id, 36).toUpperCase(Locale.ENGLISH);
28  	}
29  }
