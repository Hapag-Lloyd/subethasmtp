11  import org.subethamail.smtp.AuthenticationHandler;
12  import org.subethamail.smtp.AuthenticationHandlerFactory;
13  import org.subethamail.smtp.RejectException;
14  
15  /**
16   * This handler combines the behavior of several other authentication handler
17   * factories.
18   *
19   * @author Jeff Schnitzer
20   */
21  public class MultipleAuthenticationHandlerFactory implements AuthenticationHandlerFactory {
22  	/**
23  	 * Maps the auth type (eg "PLAIN") to a handler. The mechanism name (key) is in
24  	 * upper case.
25  	 */
26  	Map<String, AuthenticationHandlerFactory> plugins = new HashMap<>();
27  
28  	/**
29  	 * A more orderly list of the supported mechanisms. Mechanism names are in upper
30  	 * case.
31  	 */
32  	List<String> mechanisms = new ArrayList<>();
33  
34  	public MultipleAuthenticationHandlerFactory() {
35  		// Starting with an empty list is ok, let the user add them all
36  	}
37  
38  	public MultipleAuthenticationHandlerFactory(final Collection<AuthenticationHandlerFactory> factories) {
39  		for (final AuthenticationHandlerFactory fact : factories) {
40  			this.addFactory(fact);
41  		}
42  	}
43  
44  	public void addFactory(final AuthenticationHandlerFactory fact) {
45  		final List<String> partialMechanisms = fact.getAuthenticationMechanisms();
46  		for (final String mechanism : partialMechanisms) {
47  			if (!this.mechanisms.contains(mechanism)) {
48  				this.mechanisms.add(mechanism);
49  				this.plugins.put(mechanism, fact);
50  			}
51  		}
52  	}
53  
54  	@Override
55  	public List<String> getAuthenticationMechanisms() {
56  		return this.mechanisms;
57  	}
58  
59  	@Override
60  	public AuthenticationHandler create() {
61  		return new Handler();
62  	}
63  
64  	/**
65  	 */
66  	class Handler implements AuthenticationHandler {
67  		AuthenticationHandler active;
68  
69  		/* */
70  		@Override
71  		public String auth(final String clientInput) throws RejectException {
72  			if (this.active == null) {
73  				final StringTokenizer stk = new StringTokenizer(clientInput);
74  				final String auth = stk.nextToken();
75  				if (!"AUTH".equalsIgnoreCase(auth)) {
76  					throw new IllegalArgumentException("Not an AUTH command: " + clientInput);
77  				}
78  
79  				final String method = stk.nextToken();
80  				final AuthenticationHandlerFactory fact
81  						= MultipleAuthenticationHandlerFactory.this.plugins.get(method.toUpperCase(Locale.ENGLISH));
82  
83  				if (fact == null) {
84  					throw new RejectException(504, "Method not supported");
85  				}
86  
87  				this.active = fact.create();
88  			}
89  
90  			return this.active.auth(clientInput);
91  		}
92  
93  		/* */
94  		@Override
95  		public Object getIdentity() {
96  			return this.active.getIdentity();
97  		}
98  	}
99  }
