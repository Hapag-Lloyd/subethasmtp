4   import java.util.List;

7   import org.subethamail.smtp.AuthenticationHandler;

9   import org.subethamail.smtp.RejectException;

12  /**
13   * Implements the SMTP AUTH PLAIN mechanism.<br>
14   * You are only required to plug your UsernamePasswordValidator implementation
15   * for username and password validation to take effect.
16   *
17   * @author Marco Trevisan &lt;mrctrevisan@yahoo.it&gt;
18   * @author Jeff Schnitzer
19   * @author Ian White &lt;ibwhite@gmail.com&gt;
20   */
21  public class PlainAuthenticationHandlerFactory implements AuthenticationHandlerFactory {
22  	static List<String> MECHANISMS = new ArrayList<>(1);
23  	static {
24  		MECHANISMS.add("PLAIN");
25  	}
26  
27  	private final UsernamePasswordValidator helper;
28  
29  	public PlainAuthenticationHandlerFactory(final UsernamePasswordValidator helper) {
30  		this.helper = helper;
31  	}
32  
33  	@Override
34  	public List<String> getAuthenticationMechanisms() {
35  		return MECHANISMS;
36  	}
37  
38  	@Override
39  	public AuthenticationHandler create() {
40  		return new Handler();
41  	}
42  
43  	/**
44  	 */
45  	class Handler implements AuthenticationHandler {
46  		private String username;
47  
48  		private String password;
49  
50  		/* */
51  		@Override
52  		public String auth(final String clientInput) throws RejectException {
53  			final StringTokenizer stk = new StringTokenizer(clientInput);
54  			String secret = stk.nextToken();
55  			if (secret.trim().equalsIgnoreCase("AUTH")) {
56  				// Let's read the RFC2554 "initial-response" parameter
57  				// The line could be in the form of "AUTH PLAIN <base64Secret>"
58  				if (!stk.nextToken().trim().equalsIgnoreCase("PLAIN")) {
59  					// Mechanism mismatch
60  					throw new RejectException(504, "AUTH mechanism mismatch");
61  				}
62  
63  				if (!stk.hasMoreTokens()) {
64  					// the client did not submit an initial response, we'll get it in the next pass
65  					return "334 Ok";
66  				}
67  				// the client submitted an initial response
68  				secret = stk.nextToken();
69  			}
70  
71  			final byte[] decodedSecret = Base64.decode(secret);
72  			if (decodedSecret == null) {
73  				throw new RejectException(501, /* 5.5.4 */
74  						"Invalid command argument, not a valid Base64 string");
75  			}
76  
77  			/*
78  			 * RFC4616: The client presents the authorization identity (identity to act as),
79  			 * followed by a NUL (U+0000) character, followed by the authentication identity
80  			 * (identity whose password will be used), followed by a NUL (U+0000) character,
81  			 * followed by the clear-text password.
82  			 */
83  
84  			int i, j;
85  			for (i = 0; i < decodedSecret.length && decodedSecret[i] != 0; i++) {}
86  			if (i >= decodedSecret.length) {
87  				throw new RejectException(501, /* 5.5.4 */
88  						"Invalid command argument, does not contain NUL");
89  			}
90  
91  			for (j = i + 1; j < decodedSecret.length && decodedSecret[j] != 0; j++) {}
92  			if (j >= decodedSecret.length) {
93  				throw new RejectException(501, /* 5.5.4 */
94  						"Invalid command argument, does not contain the second NUL");
95  			}
96  
97  			@SuppressWarnings("unused")
98  			final String authorizationId = new String(decodedSecret, 0, i);
99  			final String authenticationId = new String(decodedSecret, i + 1, j - i - 1);
100 			final String passwd = new String(decodedSecret, j + 1, decodedSecret.length - j - 1);
101 
102 			// might be nice to do something with authorizationId, but for
103 			// purposes of the UsernamePasswordValidator, we just want to use
104 			// authenticationId
105 
106 			this.username = authenticationId;
107 			this.password = passwd;
108 			try {
109 				PlainAuthenticationHandlerFactory.this.helper.login(this.username.toString(), this.password);
110 			} catch (final LoginFailedException lfe) {
111 				throw new RejectException(535, /* 5.7.8 */
112 						"Authentication credentials invalid");
113 			}
114 
115 			return null;
116 		}
117 
118 		/* */
119 		@Override
120 		public Object getIdentity() {
121 			return this.username;
122 		}
123 	}
124 }
