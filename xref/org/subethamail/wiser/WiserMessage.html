1   package org.subethamail.wiser;
2   
3   import java.io.ByteArrayInputStream;
4   import java.io.PrintStream;
5   
6   import javax.mail.MessagingException;
7   import javax.mail.internet.MimeMessage;
8   
9   /**
10   * This class wraps a received message and provides a way to generate a JavaMail
11   * MimeMessage from the data.
12   *
13   * @author Jon Stevens
14   */
15  public class WiserMessage {
16  	byte[] messageData;
17  
18  	Wiser wiser;
19  
20  	String envelopeSender;
21  
22  	String envelopeReceiver;
23  
24  	WiserMessage(final Wiser wiser,
25  			final String envelopeSender,
26  			final String envelopeReceiver,
27  			final byte[] messageData) {
28  		this.wiser = wiser;
29  		this.envelopeSender = envelopeSender;
30  		this.envelopeReceiver = envelopeReceiver;
31  		this.messageData = messageData;
32  	}
33  
34  	/**
35  	 * Generate a JavaMail MimeMessage.
36  	 *
37  	 * @throws MessagingException on messaging error
38  	 */
39  	public MimeMessage getMimeMessage() throws MessagingException {
40  		return new MimeMessage(this.wiser.getSession(), new ByteArrayInputStream(this.messageData));
41  	}
42  
43  	/**
44  	 * Get's the raw message DATA.
45  	 */
46  	public byte[] getData() {
47  		return this.messageData;
48  	}
49  
50  	/**
51  	 * Get's the RCPT TO:
52  	 */
53  	public String getEnvelopeReceiver() {
54  		return this.envelopeReceiver;
55  	}
56  
57  	/**
58  	 * Get's the MAIL FROM:
59  	 */
60  	public String getEnvelopeSender() {
61  		return this.envelopeSender;
62  	}
63  
64  	/**
65  	 * Dumps the rough contents of the message for debugging purposes
66  	 */
67  	public void dumpMessage(final PrintStream out) throws MessagingException {
68  		out.println("===== Dumping message =====");
69  
70  		out.println("Envelope sender: " + this.getEnvelopeSender());
71  		out.println("Envelope recipient: " + this.getEnvelopeReceiver());
72  
73  		// It should all be convertible with ascii or utf8
74  		final String content = new String(this.getData());
75  		out.println(content);
76  
77  		out.println("===== End message dump =====");
78  	}
79  
80  	/**
81  	 * Implementation of toString()
82  	 *
83  	 * @return getData() as a string or an empty string if getData is null
84  	 */
85  	@Override
86  	public String toString() {
87  		if (this.getData() == null) {
88  			return "";
89  		}
90  
91  		return new String(this.getData());
92  	}
93  }
