8   import java.io.ByteArrayOutputStream;
9   import java.io.IOException;
10  import java.io.InputStream;
11  import java.io.PrintStream;
12  import java.util.ArrayList;
13  import java.util.Collections;
14  import java.util.List;
15  import java.util.Properties;

17  import javax.mail.MessagingException;
18  import javax.mail.Session;

20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  import org.subethamail.smtp.TooMuchDataException;
23  import org.subethamail.smtp.helper.SimpleMessageListener;
24  import org.subethamail.smtp.helper.SimpleMessageListenerAdapter;
25  import org.subethamail.smtp.server.SMTPServer;

27  /**
28   * Wiser is a tool for unit testing applications that send mail. Your unit tests
29   * can start Wiser, run tests which generate emails, then examine the emails
30   * that Wiser received and verify their integrity.
31   *
32   * Wiser is not intended to be a "real" mail server and is not adequate for that
33   * purpose; it simply stores all mail in memory. Use the MessageHandlerFactory
34   * interface (optionally with the SimpleMessageListenerAdapter) of SubEthaSMTP
35   * instead.
36   *
37   * @author Jon Stevens
38   * @author Jeff Schnitzer
39   */
40  public class Wiser implements SimpleMessageListener {
41  	private final static Logger log = LoggerFactory.getLogger(Wiser.class);
42  
43  	SMTPServer server;
44  
45  	protected List<WiserMessage> messages = Collections.synchronizedList(new ArrayList<WiserMessage>());
46  
47  	/**
48  	 * Create a new SMTP server with this class as the listener. The default port is
49  	 * 25. Call setPort()/setHostname() before calling start().
50  	 */
51  	public Wiser() {
52  		this.server = new SMTPServer(new SimpleMessageListenerAdapter(this));
53  	}
54  
55  	/**
56  	 * Convenience constructor
57  	 *
58  	 * @param port the port to listen on
59  	 */
60  	public Wiser(final int port) {
61  		this();
62  		this.setPort(port);
63  	}
64  
65  	/**
66  	 * The port that the server should listen on.
67  	 *
68  	 * @param port the port to listen on
69  	 */
70  	public void setPort(final int port) {
71  		this.server.setPort(port);
72  	}
73  
74  	/**
75  	 * The hostname that the server should listen on.
76  	 *
77  	 * @param hostname the hostname to listen on
78  	 */
79  	public void setHostname(final String hostname) {
80  		this.server.setHostName(hostname);
81  	}
82  
83  	/** Starts the SMTP Server */
84  	public void start() {
85  		this.server.start();
86  	}
87  
88  	/** Stops the SMTP Server */
89  	public void stop() {
90  		this.server.stop();
91  	}
92  
93  	/**
94  	 * A main() for this class. Starts up the server.
95  	 *
96  	 * @param args command line arguments
97  	 * @throws Exception on error
98  	 */
99  	public static void main(final String[] args) throws Exception {
100 		final Wiser wiser = new Wiser();
101 		wiser.start();
102 	}
103 
104 	/** Always accept everything */
105 	@Override
106 	public boolean accept(final String from, final String recipient) {
107 		if (log.isDebugEnabled()) {
108 			log.debug("Accepting mail from " + from + " to " + recipient);
109 		}
110 
111 		return true;
112 	}
113 
114 	/** Cache the messages in memory */
115 	@Override
116 	public void deliver(final String from, final String recipient, InputStream data)
117 			throws TooMuchDataException, IOException {
118 		if (log.isDebugEnabled()) {
119 			log.debug("Delivering mail from " + from + " to " + recipient);
120 		}
121 
122 		final ByteArrayOutputStream out = new ByteArrayOutputStream();
123 		data = new BufferedInputStream(data);
124 
125 		// read the data from the stream
126 		int current;
127 		while ((current = data.read()) >= 0) {
128 			out.write(current);
129 		}
130 
131 		final byte[] bytes = out.toByteArray();
132 
133 		if (log.isDebugEnabled()) {
134 			log.debug("Creating message from data with " + bytes.length + " bytes");
135 		}
136 
137 		// create a new WiserMessage.
138 		this.messages.add(new WiserMessage(this, from, recipient, bytes));
139 	}
140 
141 	/**
142 	 * Creates the JavaMail Session object for use in WiserMessage
143 	 *
144 	 * @return the JavaMail session
145 	 */
146 	protected Session getSession() {
147 		return Session.getDefaultInstance(new Properties());
148 	}
149 
150 	/**
151 	 * Returns the list of WiserMessages.
152 	 *
153 	 * <p>
154 	 * The number of mail transactions and the number of mails may be different. If
155 	 * a message is received with multiple recipients in a single mail transaction,
156 	 * then the list will contain more WiserMessage instances, one for each
157 	 * recipient.
158 	 *
159 	 * @return the list of WiserMessages
160 	 */
161 	public List<WiserMessage> getMessages() {
162 		return this.messages;
163 	}
164 
165 	/**
166 	 * Returns the server implementation
167 	 *
168 	 * @return the server implementation
169 	 */
170 	public SMTPServer getServer() {
171 		return this.server;
172 	}
173 
174 	/**
175 	 * For debugging purposes, dumps a rough outline of the messages to the output
176 	 * stream.
177 	 *
178 	 * @param out the output PrintStream
179 	 * @throws MessagingException on dump error
180 	 */
181 	public void dumpMessages(final PrintStream out) throws MessagingException {
182 		out.println("----- Start printing messages -----");
183 
184 		for (final WiserMessage wmsg : this.getMessages()) {
185 			wmsg.dumpMessage(out);
186 		}
187 
188 		out.println("----- End printing messages -----");
189 	}
190 }
