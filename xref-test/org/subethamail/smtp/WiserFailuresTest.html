173 	public void testSendTwoMessagesSameConnection() throws IOException {
174 		try {
175 			final MimeMessage[] mimeMessages = new MimeMessage[2];
176 			final Properties mailProps = this.getMailProperties(SMTP_PORT);
177 			final Session session = Session.getInstance(mailProps, null);
178 			// session.setDebug(true);
179 
180 			mimeMessages[0]
181 					= this.createMessage(session, "sender@whatever.com", "receiver@home.com", "Doodle1", "Bug1");
182 			mimeMessages[1]
183 					= this.createMessage(session, "sender@whatever.com", "receiver@home.com", "Doodle2", "Bug2");
184 
185 			final Transport transport = session.getTransport("smtp");
186 			transport.connect("localhost", SMTP_PORT, null, null);
187 
188 			for (final MimeMessage mimeMessage : mimeMessages) {
189 				transport.sendMessage(mimeMessage, mimeMessage.getAllRecipients());
190 			}
191 
192 			transport.close();
193 		} catch (final MessagingException e) {
194 			e.printStackTrace();
195 			fail("Unexpected exception: " + e);
196 		}
197 
198 		assertEquals(2, this.server.getMessages().size());
199 	}

201 	public void testSendTwoMsgsWithLogin() throws MessagingException, IOException {
202 		try {
203 			final String From = "sender@here.com";
204 			final String To = "receiver@there.com";
205 			final String Subject = "Test";
206 			final String body = "Test Body";
207 
208 			final Session session = Session.getInstance(this.getMailProperties(SMTP_PORT), null);
209 			final Message msg = new MimeMessage(session);
210 			msg.setFrom(new InternetAddress(From));
211 			msg.setRecipients(Message.RecipientType.TO, InternetAddress.parse(To, false));
212 			msg.setSubject(Subject);
213 
214 			msg.setText(body);
215 			msg.setHeader("X-Mailer", "musala");
216 			msg.setSentDate(new Date());
217 
218 			Transport transport = null;
219 
220 			try {
221 				transport = session.getTransport("smtp");
222 				transport.connect(HOST_NAME, SMTP_PORT, "ddd", "ddd");
223 				assertEquals(0, this.server.getMessages().size());
224 				transport.sendMessage(msg, InternetAddress.parse(To, false));
225 				assertEquals(1, this.server.getMessages().size());
226 				transport.sendMessage(msg, InternetAddress.parse("dimiter.bakardjiev@musala.com", false));
227 				assertEquals(2, this.server.getMessages().size());
228 			} catch (final Exception e) {
229 				e.printStackTrace();
230 			} finally {
231 				if (transport != null) {
232 					transport.close();
233 				}
234 			}
235 		} catch (final Exception e) {
236 			e.printStackTrace();
237 		}
238 
239 		final Iterator<WiserMessage> emailIter = this.server.getMessages().iterator();
240 		final WiserMessage email = emailIter.next();
241 		final MimeMessage mime = email.getMimeMessage();
242 		assertEquals("Test", mime.getHeader("Subject")[0]);
243 		assertEquals("Test Body" + "\r\n", mime.getContent().toString());
244 	}

295 	private MimeMessage createMessageWithCharset(final Session session,
296 			final String from,
297 			final String to,
298 			final String subject,
299 			final String body,
300 			final String charset) throws MessagingException {
301 		final MimeMessage msg = new MimeMessage(session);
302 		msg.setFrom(new InternetAddress(from));
303 		msg.setSubject(subject);
304 		msg.setSentDate(new Date());
305 		if (charset != null) {
306 			msg.setText(body, charset);
307 			msg.setHeader("Content-Transfer-Encoding", "7bit");
308 		} else {
309 			msg.setText(body);
310 		}
311 
312 		msg.setRecipient(Message.RecipientType.TO, new InternetAddress(to));
313 		return msg;
314 	}

363 	private void send(final String msg) throws IOException {
364 		// Force \r\n since println() behaves differently on different platforms
365 		this.output.print(msg + "\r\n");
366 		this.output.flush();
367 	}

369 	private String readInput() {
370 		final StringBuffer sb = new StringBuffer();
371 		try {
372 			do {
373 				sb.append(this.input.readLine()).append("\n");
374 			} while (this.input.ready());
375 		} catch (final IOException e) {
376 			e.printStackTrace();
377 		}
378 
379 		return sb.toString();
380 	}
