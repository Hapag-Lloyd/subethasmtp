74  	public void testReceivedHeader() throws Exception {
75  		final MimeMessage message = new MimeMessage(this.session);
76  		message.addRecipient(Message.RecipientType.TO, new InternetAddress("anyone@anywhere.com"));
77  		message.setFrom(new InternetAddress("someone@somewhereelse.com"));
78  		message.setSubject("barf");
79  		message.setText("body");
80  
81  		Transport.send(message);
82  
83  		assertEquals(1, this.wiser.getMessages().size());
84  
85  		final String[] receivedHeaders = this.wiser.getMessages().get(0).getMimeMessage().getHeader("Received");
86  
87  		assertEquals(1, receivedHeaders.length);

174 	public void testBinaryEightBitMessage() throws Exception {
175 		final byte[] body = new byte[64];
176 		new Random().nextBytes(body);
177 
178 		final MimeMessage message = new MimeMessage(this.session);
179 		message.addRecipient(Message.RecipientType.TO, new InternetAddress("anyone@anywhere.com"));
180 		message.setFrom(new InternetAddress("someone@somewhereelse.com"));
181 		message.setSubject("hello");
182 		message.setHeader("Content-Transfer-Encoding", "8bit");
183 		message.setDataHandler(new DataHandler(new ByteArrayDataSource(body, "application/octet-stream")));
184 
185 		Transport.send(message);
186 
187 		final InputStream in = this.wiser.getMessages().get(0).getMimeMessage().getInputStream();
188 		final ByteArrayOutputStream tmp = new ByteArrayOutputStream();
189 		final byte[] buf = new byte[64];
190 		int n;
191 		while ((n = in.read(buf)) != -1) {
192 			tmp.write(buf, 0, n);
193 		}
194 		in.close();
195 
196 		assertTrue(Arrays.equals(body, tmp.toByteArray()));
197 	}
