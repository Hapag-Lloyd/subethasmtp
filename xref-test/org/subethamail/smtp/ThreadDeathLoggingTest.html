14  public class ThreadDeathLoggingTest {
15  	/**
16  	 * This test can be used to check if an Error or RuntimeException actually
17  	 * logged, but it requires manual running. For example remove mail.jar from the
18  	 * classpath, and check the error log if it contains logged NoClassDef
19  	 * exception. See also the comment within the function how to check for a NPE.
20  	 * Note that any exception that causes a thread death is printed on stderr by
21  	 * the default uncaughtExceptionHandler of the JRE, but this is not what you are
22  	 * looking for.
23  	 *
24  	 * @throws SMTPException on SMTP error
25  	 * @throws IOException   on IO error
26  	 */
27  	@Ignore("Requires manual setup and verification")
28  	@Test()
29  	public void testNoMailJar() throws SMTPException, IOException {
30  		// if this variable is set to null, than a NPE will be thrown, which is
31  		// also good for testing.
32  		final MessageHandlerFactory handlerFactory = new SimpleMessageListenerAdapter(new SimpleMessageListener() {
33  			@Override
34  			public void deliver(final String from, final String recipient, final InputStream data)
35  					throws TooMuchDataException, IOException {}
36  
37  			@Override
38  			public boolean accept(final String from, final String recipient) {
39  				return false;
40  			}
41  		});
42  		final SMTPServer smtpServer = new SMTPServer(handlerFactory);
43  		smtpServer.setPort(0);
44  		smtpServer.start();
45  		try {
46  			final SmartClient client = new SmartClient("localhost", smtpServer.getPort(), "test-client.example.org");
47  			client.from("john@exmaple.com");
48  			client.to("jane@example.org");
49  			client.quit();
50  		} finally {
51  			smtpServer.stop();
52  		}
53  	}
54  }
