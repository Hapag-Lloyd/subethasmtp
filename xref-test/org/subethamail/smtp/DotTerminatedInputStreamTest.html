16  	public void testEmpty() throws IOException {
17  		final InputStream in = new ByteArrayInputStream(".\r\n".getBytes("US-ASCII"));
18  		try (DotTerminatedInputStream stream = new DotTerminatedInputStream(in)) {
19  			assertEquals(-1, stream.read());

21  	}

23  	@Test
24  	public void testPreserveLastCrLf() throws IOException {
25  		final InputStream in = new ByteArrayInputStream("a\r\n.\r\n".getBytes("US-ASCII"));
26  		final DotTerminatedInputStream stream = new DotTerminatedInputStream(in);
27  		assertEquals("a\r\n", readFull(stream));
28  	}
29  
30  	@Test
31  	public void testDotDot() throws IOException {
32  		final InputStream in = new ByteArrayInputStream("..\r\n.\r\n".getBytes("US-ASCII"));
33  		final DotTerminatedInputStream stream = new DotTerminatedInputStream(in);
34  		assertEquals("..\r\n", readFull(stream));
35  	}
36  
37  	@Test(expected = EOFException.class)
38  	public void testMissingDotLine() throws IOException {
39  		final InputStream in = new ByteArrayInputStream("a\r\n".getBytes("US-ASCII"));
40  		final DotTerminatedInputStream stream = new DotTerminatedInputStream(in);
41  		readFull(stream);
42  	}
43  
44  	private String readFull(final DotTerminatedInputStream in) throws IOException {
45  		final ByteArrayOutputStream out = new ByteArrayOutputStream();
46  		int ch;
47  		while (-1 != (ch = in.read())) {
48  			out.write(ch);
49  		}
50  		return out.toString("US-ASCII");
51  	}
