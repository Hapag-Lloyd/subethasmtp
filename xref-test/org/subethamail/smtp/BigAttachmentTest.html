76  		if (BIGFILE_PATH.equals(TO_CHANGE)) {
77  			log.error(
78  					"BigAttachmentTest: To complete this test you must change the BIGFILE_PATH var to point out a file on your disk !");
79  		}
80  		assertNotSame(
81  				"BigAttachmentTest: To complete this test you must change the BIGFILE_PATH var to point out a file on your disk !",
82  				TO_CHANGE,
83  				BIGFILE_PATH);
84  		final Properties props = System.getProperties();
85  		props.setProperty("mail.smtp.host", "localhost");
86  		props.setProperty("mail.smtp.port", SMTP_PORT + "");
87  		final Session session = Session.getInstance(props);
88  
89  		final MimeMessage baseMsg = new MimeMessage(session);
90  		final MimeBodyPart bp1 = new MimeBodyPart();
91  		bp1.setHeader("Content-Type", "text/plain");
92  		bp1.setContent("Hello World!!!", "text/plain; charset=\"ISO-8859-1\"");
93  
94  		// Attach the file
95  		final MimeBodyPart bp2 = new MimeBodyPart();
96  		final FileDataSource fileAttachment = new FileDataSource(BIGFILE_PATH);
97  		final DataHandler dh = new DataHandler(fileAttachment);
98  		bp2.setDataHandler(dh);
99  		bp2.setFileName(fileAttachment.getName());
100 
101 		final Multipart multipart = new MimeMultipart();
102 		multipart.addBodyPart(bp1);
103 		multipart.addBodyPart(bp2);
104 
105 		baseMsg.setFrom(new InternetAddress("Ted <ted@home.com>"));
106 		baseMsg.setRecipient(Message.RecipientType.TO, new InternetAddress("success@subethamail.org"));
107 		baseMsg.setSubject("Test Big attached file message");
108 		baseMsg.setContent(multipart);
109 		baseMsg.saveChanges();
110 
111 		log.debug("Send started");
112 		final Transport t = new SMTPTransport(session, new URLName("smtp://localhost:" + SMTP_PORT));
113 		long started = System.currentTimeMillis();
114 		t.connect();
115 		t.sendMessage(baseMsg, new Address[] { new InternetAddress("success@subethamail.org") });
116 		t.close();
117 		started = System.currentTimeMillis() - started;
118 		log.info("Elapsed ms = " + started);
119 
120 		final WiserMessage msg = this.server.getMessages().get(0);
121 
122 		assertEquals(1, this.server.getMessages().size());
123 		assertEquals("success@subethamail.org", msg.getEnvelopeReceiver());
124 
125 		final File compareFile = File.createTempFile("attached", ".tmp");
126 		log.debug("Writing received attachment ...");
127 
128 		final FileOutputStream fos = new FileOutputStream(compareFile);
129 		((MimeMultipart) msg.getMimeMessage().getContent()).getBodyPart(1).getDataHandler().writeTo(fos);
130 		fos.close();
131 		log.debug("Checking integrity ...");
132 		assertTrue(this.checkIntegrity(new File(BIGFILE_PATH), compareFile));
133 		log.debug("Checking integrity DONE");

137 	private boolean checkIntegrity(final File src, final File dest) throws IOException, NoSuchAlgorithmException {
138 		final BufferedInputStream ins = new BufferedInputStream(new FileInputStream(src));
139 		final BufferedInputStream ind = new BufferedInputStream(new FileInputStream(dest));
140 		final MessageDigest md1 = MessageDigest.getInstance("MD5");
141 		final MessageDigest md2 = MessageDigest.getInstance("MD5");
142 
143 		int r = 0;
144 		final byte[] buf1 = new byte[BUFFER_SIZE];
145 		final byte[] buf2 = new byte[BUFFER_SIZE];
146 
147 		while (r != -1) {
148 			r = ins.read(buf1);
149 			ind.read(buf2);
150 
151 			md1.update(buf1);
152 			md2.update(buf2);
153 		}
154 
155 		ins.close();
156 		ind.close();
157 		return MessageDigest.isEqual(md1.digest(), md2.digest());
158 	}
